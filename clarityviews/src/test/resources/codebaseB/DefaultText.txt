package codebaseA;

public class DefaultText implements Text {

    private String text;
    String color;

    public DefaultText(String text, String color) {
        this.text = text;
        this.color = color;
    }

    @Override
    public String value() {
        return this.text;
    }
    
    @Override
    public String color() {
        return this.color;
    }
}
